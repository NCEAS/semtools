
# TEST conjunctive conditions with aggregation
################################################
#natural form of queries
#Q1: %Tree%['DBH' > 35.00 && count('DBH') <=1]
#
# Result: 
# [(1,6), (2,5), (2,9)]
# tb1:  
# 6 |    2 | abba             | 2008 | 36.5
#		 tb2: 
# SELECT DISTINCT tree,spp,plot FROM tb2 WHERE dbh > 35.0 GROUP BY tree,spp,plot HAVING count(*)<=1                                                     ;
# tree |       spp        |   plot   
#------+------------------+----------
#    2 | pine tree        | B       
#    2 | abba             | B
# 
BEGIN QUERY:1
BEGIN QO:1
EntityId: '%Tree%'
BEGIN QM:1
Characteristic:'DBH'
Standard:
Cond:>35.0
Aggregation:
CNF:1
END QM
######## AND measurement type, these two measurements are in on CNF 
BEGIN QM:2
Characteristic:'DBH'
Standard:
Cond:<=1
Aggregation:count
CNF:1
END QM
END QO
END QUERY
